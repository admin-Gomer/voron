echo "============================================================"
echo "💎 УСТАНОВКА ЗАВИСИМОСТЕЙ ДЛЯ БОТА"

# Обновление пакетов
echo "📦 Обновление пакетов..."
pkg update && pkg upgrade -y

# Установка Python
echo "🐍 Установка Python..."
pkg install python python-pip -y

# Установка библиотек
echo "📚 Установка библиотек Python..."
pip install python-telegram-bot Pillow aiohttp

# Проверка
echo "============================================================"
echo "✅ ПРОВЕРКА УСТАНОВКИ"
echo "============================================================"
python -c "import telegram; import PIL; import aiohttp; print('✅ Все библиотеки установлены успешно!')"

echo "============================================================"
echo "🚀 Теперь запусти бота: python bot.py"
echo "============================================================"

~/voron-main/
│
├── 📄 bot.py                    # Главный файл бота (основной код)
│
├── 📁 admin_media/              # Временные медиафайлы (создаётся автоматически)
│   ├── *.jpg                    # Временные фото для постов
│   ├── *.mp4                    # Временные видео для постов
│   ├── *.gif                    # Временные GIF для постов
│   ├── welcome_*.png            # Временные приветствия
│   ├── goodbye_*.png            # Временные прощания
│   └── watermarked_*.png        # Временные фото с водяным знаком
│
├── 📁 watermarks/               # Папка для аватарок водяного знака
│   ├── custom_avatar.png        # Твоя кастомная аватарка (если есть)
│   └── default_avatar.png       # Дефолтная аватарка (создаётся автоматически)
│
├── 📁 fonts/                    # Папка для шрифтов (опционально)
│   ├── Montserrat-Bold.ttf      # Шрифт для водяного знака
│   ├── Roboto-Bold.ttf          # Альтернативный шрифт
│   └── arialbd.ttf              # Стандартный шрифт
│
├── 📁 default_avatars/          # Папка для аватарок-заглушек (если у пользователя нет аватарки)
│   ├── avatar1.jpg              # Твои картинки (любое имя)
│   ├── avatar2.png              # Поддерживаются: jpg, png, webp
│   ├── avatar3.gif              # GIF тоже работают
│   └── default_placeholder.png  # Создаётся автоматически, если папка пустая
│
├── 📄 file_db.json              # База данных файлов (ID, file_id, имя файла)
├── 📄 group_ids.json            # Список групп, где работает бот (если ENABLE_ALL_GROUPS = False)
│
└── 📄 donations_db.json         # (Удалён, больше не нужен)
